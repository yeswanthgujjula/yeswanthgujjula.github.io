This project predicts the driving range of electric vehicles and finds the optimal route with charging stops, using machine learning and the A* search algorithm.

import heapq

# A* algorithm for EV route planning
def astar(start, goal, graph, battery_range):
    open_set = []
    heapq.heappush(open_set, (0, start, battery_range))
    came_from = {}
    g_score = {start: 0}

    while open_set:
        _, current, remaining = heapq.heappop(open_set)

        if current == goal:
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return path[::-1]

        for neighbor, dist in graph.get(current, []):
            if dist > remaining:
                continue  # can't reach without charging
            tentative_g = g_score[current] + dist
            if tentative_g < g_score.get(neighbor, float('inf')):
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score, neighbor, battery_range - dist))

def heuristic(a, b):
    # simple straight-line estimate
    return abs(a - b)

graph = {
    0: [(1, 5), (2, 10)],
    1: [(2, 3), (3, 7)],
    2: [(3, 2)],
    3: []
}

print(astar(0, 3, graph, battery_range=10))
